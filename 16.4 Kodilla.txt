SELECT I.ID AS ISSUE_ID, I.SUMMARY AS ISSUE_SUMMARY, I.DESCRIPTION AS ISSUE_DESCRIPTION, IL.NAME AS LIST_NAME
FROM ISSUES AS I
        JOIN ISSUESLISTS AS IL ON I.ISSUESLIST_ID = IL.ID;

SELECT I.ID AS ISSUE_ID, I.SUMMARY AS ISSUE_SUMMARY, I.DESCRIPTION AS ISSUE_DESCRIPTION, U.FIRSTNAME AS USER_FIRSTNAME, U.LASTNAME AS USER_LASTNAME
FROM ISSUES AS I
         LEFT JOIN USERS AS U ON I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME AS USER_FIRSTNAME, U.LASTNAME AS USER_LASTNAME, COUNT(I.ID) AS TASK_COUNT
FROM USERS AS U
         LEFT JOIN ISSUES AS I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.FIRSTNAME, U.LASTNAME;
